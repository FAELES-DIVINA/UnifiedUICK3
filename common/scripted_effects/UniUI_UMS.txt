UMS_toggle_mod = {
	if = {
		limit = { 
			NOT = {exists = global_var:$MOD$_is_loaded}
		}
		set_global_variable = {
			name = $MOD$_is_loaded
			value = yes
		}
	}

	else = {
		remove_global_variable = $MOD$_is_loaded
	}
}
